using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace contoh_pengulangan_for
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Mulai perulangan untuk menampilkan angka 1 sampai 10:");

            // Perulangan for untuk menampilkan angka dari 1 hingga 10
            for (int i = 1; i <= 10; i++)
            {
                Console.WriteLine($"Angka ke: {i}"); // Menggunakan interpolasi string
            }

            Console.WriteLine("Perulangan selesai.\n");
        }
    }
}
